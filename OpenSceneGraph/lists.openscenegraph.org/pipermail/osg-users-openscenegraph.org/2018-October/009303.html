Code:

      
        rig.getSourceGeometry()->setStateSet( (osg::StateSet*)  rig.getSourceGeometry()->getStateSet()->clone(osg::CopyOp::SHALLOW_COPY));
        // copy shallow from source geometry to rig
        rig.copyFrom(source);
