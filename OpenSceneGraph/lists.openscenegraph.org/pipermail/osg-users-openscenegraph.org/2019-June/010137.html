SET(TARGET_SRC osgviewer.cpp )

        OpenThreads
        osg
        osgDB
        osgUtil
        osgGA
        osgViewer
        osgText
)

SETUP_APPLICATION(osgviewer)

TARGET_LINK_LIBRARIES(${TARGET_TARGETNAME} EGL GLdispatch OpenGL)
