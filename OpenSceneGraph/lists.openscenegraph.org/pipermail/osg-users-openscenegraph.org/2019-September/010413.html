virtual void operator()(osg::Node*, osg::NodeVisitor* nv)
{
	if (nv->getFrameStamp())

		_coord = dynamic_cast<osg::Vec2Array*>(_geom->getTexCoordArray(0));
			
		if (_coord)

			{
