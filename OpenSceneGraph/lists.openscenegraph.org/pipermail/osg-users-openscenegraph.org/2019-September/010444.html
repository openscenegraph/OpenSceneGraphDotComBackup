virtual void operator()(osg::Node*, osg::NodeVisitor* nv)
{		
	if (nv->getFrameStamp())
	{		
		if (_coord)
		{
			for (unsigned int u = 0; u < _coord->size(); u++)
			{
				_coord->at(u).x() += 0.6f;
				_coord->at(u).y() += 0.6f;
				_coord->dirty();
			}
			_geom->setUseVertexBufferObjects(true);
			_coord->getBufferObject()->dirty();
		}
	}
}
