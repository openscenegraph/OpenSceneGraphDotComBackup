I managed to do that inside the `handle` method:

    osg::Vec3d eyePos;
    osg::Matrix matrix = _viewer->getCameraManipulator()->getMatrix();
    eyePos = matrix.getTrans();
    lat = eyePos.y();
    lon = eyePos.x();
    alt = eyePos.z();
